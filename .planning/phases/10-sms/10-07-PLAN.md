    기존 `ThrottlerModule.forRoot([{ ttl: 60000, limit: 60 }])` 를 제거하고:
    ```typescript
    ThrottlerModule.forRootAsync({
      imports: [BookingModule],   // REDIS_CLIENT re-export
      inject: [REDIS_CLIENT],
      useFactory: (redis: IORedis) => {
        // [RESEARCH Pitfall 5] InMemoryRedis는 incr 메서드 미구현 → storage omit (dev fallback)
        // 실 ioredis는 incr로 INCR 명령 실행 가능 → Valkey storage 연결
        // typeof redis.call 대신 typeof redis.incr 사용: RESEARCH §Pitfall 5 기준 일치
        const isRealRedis = typeof (redis as IORedis).incr === 'function';
        return {
          // [Review #6] v6 ms 단위: 60_000ms = 1분 global default
          throttlers: [{ name: 'default', ttl: 60_000, limit: 60 }],
          ...(isRealRedis
            ? { storage: new ThrottlerStorageRedisService(redis) }
            : {}),  // dev: in-memory throttler fallback
        };
      },
    }),
    ```
