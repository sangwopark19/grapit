    echo "  D-09: gh variable set CLOUD_RUN_API_URL --body 'https://api.heygrabit.com'"

    echo "=== [4/5] Smoke — new Cloud Run services (run AFTER deploy.yml PR merge) ==="
    for svc in grabit-api grabit-web; do
