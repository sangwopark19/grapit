    STEP 1 — `scripts/rollback-cutover.sh` 작성 (기존 Revision 1 내용 유지, apex 대상 명시):
    ```bash
    #!/usr/bin/env bash
    # scripts/rollback-cutover.sh
    # Phase 13 Plan 04 — Rollback the heygrabit.com apex domain mapping cutover.
    # Note: api.heygrabit.com 서브도메인은 Plan 03 에서 grabit-api 로 고정됨 — 본 스크립트는 apex (heygrabit.com) 만 대상.
    # Usage:
    #   1. BEFORE cutover: ./scripts/rollback-cutover.sh capture       # captures rollback.yaml
    #   2. AFTER cutover (if needed): ./scripts/rollback-cutover.sh restore
    # Note: rollback.yaml is git-ignored (see .gitignore) — do not commit.
    set -euo pipefail

    DOMAIN="heygrabit.com"
    REGION="asia-northeast3"
    OLD_SERVICE="grapit-web"
    ROLLBACK_FILE="rollback.yaml"

    case "${1:-}" in
      capture)
        echo "=== Capturing current apex domain mapping -> $ROLLBACK_FILE ==="
        gcloud beta run domain-mappings describe \
          --domain="$DOMAIN" \
          --region="$REGION" \
          --format=yaml > "$ROLLBACK_FILE"
        if grep -q "routeName: $OLD_SERVICE" "$ROLLBACK_FILE"; then
          echo "  Captured mapping for $OLD_SERVICE — rollback path ready"
        else
          echo "  WARNING: expected routeName=$OLD_SERVICE not found in rollback.yaml"
          echo "  Current mapping:"
          grep routeName "$ROLLBACK_FILE" || true
        fi
        ;;
      restore)
        echo "=== Restoring apex domain mapping to $OLD_SERVICE ==="
        [ -f "$ROLLBACK_FILE" ] || { echo "ERROR: $ROLLBACK_FILE not found"; exit 1; }
        gcloud beta run domain-mappings delete \
          --domain="$DOMAIN" \
          --region="$REGION" \
          --quiet || true
        gcloud beta run domain-mappings create \
          --service="$OLD_SERVICE" \
          --domain="$DOMAIN" \
          --region="$REGION"
        for i in $(seq 1 60); do
          STATUS=$(gcloud beta run domain-mappings describe --domain="$DOMAIN" --region="$REGION" --format='value(status.conditions[0].status)' 2>/dev/null || echo "Unknown")
          echo "  [$i/60] status=$STATUS"
          [ "$STATUS" = "True" ] && { echo "  Rollback ready"; break; }
          sleep 15
        done
        curl -sI "https://$DOMAIN/" | head -1 || true
        echo "=== Apex rollback complete (api.heygrabit.com 서브도메인은 영향 없음) ==="
        ;;
      *)
        echo "Usage: $0 {capture|restore}"
        echo "  capture: save current apex mapping as rollback.yaml (run BEFORE cutover)"
        echo "  restore: revert apex mapping to $OLD_SERVICE (run AFTER failed cutover)"
        exit 1
        ;;
    esac
    ```

    ```bash
    # D-14 Hard gate (3 preconditions):
    #   1. --confirm-after-date required; today must be >= argument date
    #   2. domain-mappings: heygrabit.com → grabit-web AND api.heygrabit.com → grabit-api
    #   3. 24h traffic log for grapit-{web,api} must be empty
    set -euo pipefail

    usage() {
      echo "Usage: $0 <GCP_PROJECT_ID> --confirm-after-date YYYY-MM-DD"
      echo "  D-14: date argument is MANDATORY. Today must be >= that date."
      exit 1
    }

    PROJECT_ID="${1:-}"
    shift || true
    [ -z "$PROJECT_ID" ] && usage
    CONFIRM_DATE=""
    while [ $# -gt 0 ]; do
      case "$1" in
        --confirm-after-date)
          CONFIRM_DATE="${2:-}"; shift 2 ;;
        *) echo "Unknown argument: $1"; usage ;;
      esac
    done
    [ -z "$CONFIRM_DATE" ] && { echo "ERROR: --confirm-after-date YYYY-MM-DD required (D-14 hard gate)"; usage; }

    REGION="asia-northeast3"
    TODAY=$(date +%Y-%m-%d)

    echo "=== D-14 Hard Gate [1/3] — date precondition ==="
    # Lexicographic compare works for YYYY-MM-DD format
    if [[ "$TODAY" < "$CONFIRM_DATE" ]]; then
      echo "ERROR: today ($TODAY) < --confirm-after-date ($CONFIRM_DATE). Refuse to run." >&2
      exit 1
    fi
    echo "  OK: today=$TODAY >= confirm-after-date=$CONFIRM_DATE"

    echo "=== D-14 Hard Gate [2/3] — domain-mapping routeName check (apex + api) ==="
    APEX_ROUTE=$(gcloud beta run domain-mappings describe --domain=heygrabit.com --region="$REGION" --project="$PROJECT_ID" --format='value(spec.routeName)' 2>/dev/null || echo "")
    if [ "$APEX_ROUTE" != "grabit-web" ]; then
      echo "ERROR: heygrabit.com routeName expected 'grabit-web', got '$APEX_ROUTE'. Refuse to run." >&2
      exit 1
    fi
    echo "  OK: heygrabit.com → grabit-web"

    API_ROUTE=$(gcloud beta run domain-mappings describe --domain=api.heygrabit.com --region="$REGION" --project="$PROJECT_ID" --format='value(spec.routeName)' 2>/dev/null || echo "")
    if [ "$API_ROUTE" != "grabit-api" ]; then
      echo "ERROR: api.heygrabit.com routeName expected 'grabit-api', got '$API_ROUTE'. Refuse to run." >&2
      exit 1
    fi
    echo "  OK: api.heygrabit.com → grabit-api"

    echo "=== D-14 Hard Gate [3/3] — 24h traffic log empty check ==="
    LAST_TS=$(gcloud logging read \
      'resource.type="cloud_run_revision" AND resource.labels.service_name=("grapit-web" OR "grapit-api")' \
      --project="$PROJECT_ID" \
      --freshness=24h \
      --limit=1 \
      --format='value(timestamp)' 2>/dev/null || echo "")
    if [ -n "$LAST_TS" ]; then
      echo "ERROR: grapit-* received traffic within last 24h (last: $LAST_TS). Refuse to run." >&2
      exit 1
    fi
    echo "  OK: no grapit-* traffic in last 24h"

    echo "=== All D-14 hard gates passed. Proceeding with cleanup. ==="

    echo "=== [1/3] Delete old Cloud Run services ==="
    for svc in grapit-web grapit-api; do
      if gcloud run services describe "$svc" --region="$REGION" --project="$PROJECT_ID" >/dev/null 2>&1; then
        echo "  Deleting $svc..."
        gcloud run services delete "$svc" --region="$REGION" --project="$PROJECT_ID" --quiet
      else
        echo "  $svc not found (already deleted?) — skip"
      fi
    done

    echo "=== [2/3] Delete old Artifact Registry repo 'grapit' ==="
    if gcloud artifacts repositories describe grapit --location="$REGION" --project="$PROJECT_ID" >/dev/null 2>&1; then
      echo "  Deleting AR repo grapit (includes all images)..."
      gcloud artifacts repositories delete grapit --location="$REGION" --project="$PROJECT_ID" --quiet
    else
      echo "  AR repo grapit not found — skip"
    fi

    echo "=== [3/3] Sentry old projects (manual, Sentry has no gcloud-equivalent CLI) ==="
    echo "  Open Sentry dashboard → grapit-web project → Settings → General → Remove Project"
    echo "  Open Sentry dashboard → grapit-api project → Settings → General → Remove Project"
    echo "  (or use Sentry CLI: sentry-cli projects remove {org} grapit-{web,api})"

    echo "=== Cleanup complete (--confirm-after-date=$CONFIRM_DATE) ==="
    echo "Verify:"
    echo "  gcloud run services list --region=$REGION --project=$PROJECT_ID | grep grapit-  # expected: empty"
    echo "  gcloud artifacts repositories list --location=$REGION --project=$PROJECT_ID | grep grapit  # expected: empty"
    ```
