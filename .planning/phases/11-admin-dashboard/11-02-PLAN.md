    **A) `apps/api/src/modules/admin/kst-boundary.ts` 신규 작성**

    Node 측에서 Asia/Seoul 자정 경계를 UTC Date로 계산. DB 쿼리에서는 raw 컬럼과 이 Date를 바로 비교 → createdAt index 사용 가능 (review MEDIUM 4).

    ```typescript
    const KST_OFFSET_MS = 9 * 60 * 60 * 1000;

    /**
     * Returns UTC Date boundaries for "from N days ago 00:00 KST" to "tomorrow 00:00 KST".
     * Use: `reservations.createdAt >= startUtc AND reservations.createdAt < endUtc`
     * (raw column comparison → index eligible).
     */
    export function kstBoundaryToUtc(days: number): { startUtc: Date; endUtc: Date } {
      const nowUtcMs = Date.now();
      const nowKstMs = nowUtcMs + KST_OFFSET_MS;
      // KST 기준 자정(다음날 00:00 KST) = endUtc
      const kstStartOfDayMs = Math.floor(nowKstMs / 86_400_000) * 86_400_000;
      const kstEndOfTodayMs = kstStartOfDayMs + 86_400_000;
      // startUtc = (today KST 00:00) - days * 24h
      const kstStartMs = kstEndOfTodayMs - days * 86_400_000;
      return {
        startUtc: new Date(kstStartMs - KST_OFFSET_MS),
        endUtc: new Date(kstEndOfTodayMs - KST_OFFSET_MS),
      };
    }

    /**
     * "오늘 KST 00:00" ~ "내일 KST 00:00" UTC boundary.
     */
    export function kstTodayBoundaryUtc(): { startUtc: Date; endUtc: Date } {
      return kstBoundaryToUtc(0);
    }

    /**
     * YYYY-MM-DD bucket list in KST for the last `days` days, inclusive of today.
     * Returned ASC by date.
     */
    export function buildDailyBucketSkeleton(days: number): string[] {
      const nowKstMs = Date.now() + KST_OFFSET_MS;
      const todayStart = Math.floor(nowKstMs / 86_400_000) * 86_400_000;
      const buckets: string[] = [];
      for (let i = days - 1; i >= 0; i -= 1) {
        const d = new Date(todayStart - i * 86_400_000 - KST_OFFSET_MS);
        // format YYYY-MM-DD in KST
        const kst = new Date(d.getTime() + KST_OFFSET_MS);
        const y = kst.getUTCFullYear();
        const m = String(kst.getUTCMonth() + 1).padStart(2, '0');
        const day = String(kst.getUTCDate()).padStart(2, '0');
        buckets.push(`${y}-${m}-${day}`);
      }
      return buckets;
    }

    /**
     * ISO week bucket list (e.g. "2026-W17") for the last `weeks` weeks in KST.
     */
    export function buildWeeklyBucketSkeleton(weeks: number): string[] {
      const nowKstMs = Date.now() + KST_OFFSET_MS;
      const buckets: string[] = [];
      for (let i = weeks - 1; i >= 0; i -= 1) {
        const d = new Date(nowKstMs - i * 7 * 86_400_000 - KST_OFFSET_MS);
        const kst = new Date(d.getTime() + KST_OFFSET_MS);
        // ISO week calculation
        const target = new Date(Date.UTC(kst.getUTCFullYear(), kst.getUTCMonth(), kst.getUTCDate()));
        const dayNum = target.getUTCDay() || 7;
        target.setUTCDate(target.getUTCDate() + 4 - dayNum);
        const yearStart = new Date(Date.UTC(target.getUTCFullYear(), 0, 1));
        const weekNum = Math.ceil(((target.getTime() - yearStart.getTime()) / 86_400_000 + 1) / 7);
        buckets.push(`${target.getUTCFullYear()}-W${String(weekNum).padStart(2, '0')}`);
      }
      return buckets;
    }
    ```

    **B) `apps/api/src/modules/admin/admin-dashboard.service.ts` 교체 (skeleton → 실구현)**

    Plan 01 skeleton의 `throw new Error('Not implemented')` 본문을 모두 실제 Drizzle 쿼리 + readThrough cache + bucket skeleton merge로 교체. 전체 골격:

    ```typescript
    import { Injectable, Inject } from '@nestjs/common';
    import { sql, eq, inArray, and, gte, lt } from 'drizzle-orm';
    import type {
      DashboardSummaryDto,
      DashboardRevenueDto,
      DashboardGenreDto,
      DashboardPaymentDto,
      DashboardTopDto,
      DashboardPeriod,
    } from '@grapit/shared';
    import { DRIZZLE, type DrizzleDB } from '../../database/drizzle.provider.js';
    import {
      reservations,
      payments,
      performances,
      showtimes,
    } from '../../database/schema/index.js';
    import { CacheService } from '../performance/cache.service.js';
    import {
      kstBoundaryToUtc,
      kstTodayBoundaryUtc,
      buildDailyBucketSkeleton,
      buildWeeklyBucketSkeleton,
    } from './kst-boundary.js';

    const DASHBOARD_CACHE_TTL = 60; // D-12: always 60s, never omit
    // D-10: Top 10 는 고정 30일 윈도우

    @Injectable()
    export class AdminDashboardService {
      constructor(
        @Inject(DRIZZLE) private readonly db: DrizzleDB,
        private readonly cache: CacheService,
      ) {}

      private async readThrough<T>(key: string, fetcher: () => Promise<T>): Promise<T> {
        const cached = await this.cache.get<T>(key);
        if (cached !== null) return cached;
        const fresh = await fetcher();
        await this.cache.set(key, fresh, DASHBOARD_CACHE_TTL); // D-12: 60 explicit
        return fresh;
      }

      // ADM-01 — 오늘 KPI 4종
      async getSummary(): Promise<DashboardSummaryDto> {
        return this.readThrough('cache:admin:dashboard:summary', async () => {
          // review MEDIUM 4: Node 측에서 UTC boundary pre-compute → raw createdAt 비교 (index 활용 가능)
          const { startUtc, endUtc } = kstTodayBoundaryUtc();
          const todayCreatedAt = and(
            gte(reservations.createdAt, startUtc),
            lt(reservations.createdAt, endUtc),
          );

          const [bookings, revenue, cancelled, active] = await Promise.all([
            this.db.select({ count: sql<number>`count(*)::int` })
              .from(reservations)
              .where(todayCreatedAt),
            this.db.select({ sum: sql<number>`coalesce(sum(${reservations.totalAmount}), 0)::int` })
              .from(reservations)
              // D-07: revenue는 CONFIRMED만
              .where(and(todayCreatedAt, eq(reservations.status, 'CONFIRMED'))),
            this.db.select({ count: sql<number>`count(*)::int` })
              .from(reservations)
              .where(and(todayCreatedAt, eq(reservations.status, 'CANCELLED'))),
            // Pitfall 6 ("활성 공연" 정의): status IN ('selling','closing_soon')
            this.db.select({ count: sql<number>`count(*)::int` })
              .from(performances)
              .where(inArray(performances.status, ['selling', 'closing_soon'])),
          ]);

          return {
            todayBookings: bookings[0]?.count ?? 0,
            todayRevenue: revenue[0]?.sum ?? 0,
            todayCancelled: cancelled[0]?.count ?? 0,
            activePerformances: active[0]?.count ?? 0,
          };
        });
      }

      // ADM-02 — 매출 추이 (일별/주별 + 빈 bucket 0으로 채움)
      async getRevenueTrend(period: DashboardPeriod): Promise<DashboardRevenueDto> {
        return this.readThrough(`cache:admin:dashboard:revenue:${period}`, async () => {
          // D-09: 7d/30d = day, 90d = week
          const granularity: 'day' | 'week' = period === '90d' ? 'week' : 'day';
          const days = period === '7d' ? 7 : period === '30d' ? 30 : 90;
          const { startUtc, endUtc } = kstBoundaryToUtc(days);

          // granularity는 validated enum에서 derive된 literal — sql.raw 안전 (Pitfall 8)
          const granularityLiteral = sql.raw(`'${granularity}'`);
          // bucket expression: date_trunc는 createdAt을 Asia/Seoul 로컬로 변환 후 truncate
          // (결과 라벨링 용도. WHERE 절은 여전히 raw createdAt 비교로 index 활용.)
          const bucketExpr = sql`date_trunc(${granularityLiteral}, ${reservations.createdAt} AT TIME ZONE 'Asia/Seoul')`;
          const bucketFormat = granularity === 'week'
            ? sql.raw(`'IYYY-"W"IW'`)
            : sql.raw(`'YYYY-MM-DD'`);

          const rows = await this.db
            .select({
              bucket: sql<string>`to_char(${bucketExpr}, ${bucketFormat})`,
              revenue: sql<number>`coalesce(sum(${reservations.totalAmount}), 0)::int`,
              count: sql<number>`count(*)::int`,
            })
            .from(reservations)
            // review MEDIUM 4: WHERE는 raw createdAt 비교 (index eligible)
            .where(and(
              eq(reservations.status, 'CONFIRMED'),
              gte(reservations.createdAt, startUtc),
              lt(reservations.createdAt, endUtc),
            ))
            .groupBy(bucketExpr)
            .orderBy(bucketExpr);

          // review MEDIUM 6: bucket skeleton으로 빈 날짜/주 0 채움
          const skeleton = granularity === 'week'
            ? buildWeeklyBucketSkeleton(Math.ceil(days / 7))
            : buildDailyBucketSkeleton(days);
          const map = new Map(rows.map((r) => [r.bucket, r]));
          return skeleton.map((b) => map.get(b) ?? { bucket: b, revenue: 0, count: 0 });
        });
      }

      // ADM-03 — 장르별 분포 (period-shared per D-11)
      async getGenreDistribution(period: DashboardPeriod): Promise<DashboardGenreDto> {
        return this.readThrough(`cache:admin:dashboard:genre:${period}`, async () => {
          const days = period === '7d' ? 7 : period === '30d' ? 30 : 90;
          const { startUtc, endUtc } = kstBoundaryToUtc(days);
          const rows = await this.db
            .select({
              genre: sql<string>`${performances.genre}::text`,
              count: sql<number>`count(${reservations.id})::int`,
            })
            .from(reservations)
            .innerJoin(showtimes, eq(reservations.showtimeId, showtimes.id))
            .innerJoin(performances, eq(showtimes.performanceId, performances.id))
            .where(and(
              eq(reservations.status, 'CONFIRMED'),
              gte(reservations.createdAt, startUtc),
              lt(reservations.createdAt, endUtc),
            ))
            .groupBy(performances.genre)
            .orderBy(sql`count(${reservations.id}) desc`);
          return rows;
        });
      }

      // ADM-05 — 결제수단 분포 (CONFIRMED reservation + DONE payment, review MEDIUM 5 / Pitfall 5)
      async getPaymentDistribution(period: DashboardPeriod): Promise<DashboardPaymentDto> {
        return this.readThrough(`cache:admin:dashboard:payment:${period}`, async () => {
          const days = period === '7d' ? 7 : period === '30d' ? 30 : 90;
          const { startUtc, endUtc } = kstBoundaryToUtc(days);
          const rows = await this.db
            .select({
              method: payments.method,
              count: sql<number>`count(*)::int`,

            .where(and(

              lt(reservations.createdAt, endUtc),
            ))

            .orderBy(sql`count(*) desc`);
          return rows;
        });

          const rows = await this.db
            .select({
              performanceId: performances.id,
              title: performances.title,
              genre: sql<string>`${performances.genre}::text`,
              posterUrl: performances.posterUrl,
              bookingCount: sql<number>`count(${reservations.id})::int`,
            })

            .innerJoin(showtimes, eq(reservations.showtimeId, showtimes.id))

              eq(reservations.status, 'CONFIRMED'),
              gte(reservations.createdAt, startUtc),
              lt(reservations.createdAt, endUtc),
            ))
            .groupBy(performances.id, performances.title, performances.genre, performances.posterUrl)
            .orderBy(sql`count(${reservations.id}) desc`)
            .limit(10);
          return rows;
        });
      }
    }
    ```
