    ```typescript
      return useQuery({
        queryKey: ['showtimes', performanceId],
        queryFn: () => apiClient.get<Showtime[]>(`/api/v1/performances/${performanceId}/showtimes`),

      });

    export function useSeatStatus(showtimeId: string | null) {
      return useQuery({
        queryKey: ['seat-status', showtimeId],
        queryFn: () => apiClient.get<SeatStatusResponse>(`/api/v1/booking/schedules/${showtimeId}/seats`),
        enabled: !!showtimeId,
      });
    }

    export function useLockSeat() {
      const queryClient = useQueryClient();
      return useMutation({
        mutationFn: (data: LockSeatRequest) =>
          apiClient.post<LockSeatResponse>('/api/v1/booking/seats/lock', data),
        onSuccess: (_, variables) => {
          queryClient.invalidateQueries({ queryKey: ['seat-status', variables.showtimeId] });
        },
      });
    }

    export function useUnlockSeat() {
      const queryClient = useQueryClient();
      return useMutation({
        mutationFn: ({ showtimeId, seatId }: { showtimeId: string; seatId: string }) =>
          apiClient.delete<void>(`/api/v1/booking/seats/lock/${showtimeId}/${seatId}`),
        onSuccess: (_, variables) => {
          queryClient.invalidateQueries({ queryKey: ['seat-status', variables.showtimeId] });
        },
      });
    }
    ```
